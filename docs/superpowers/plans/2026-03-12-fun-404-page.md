# Fun 404 Page Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the minimal 404 page with an interactive AI terminal typewriter that promotes the Prompt Contracts book.

**Architecture:** Single-file Astro page (`404.astro`) using `BaseLayout`. HTML contains all terminal lines as static content (no-JS fallback). A vanilla JS `<script>` hides lines on load then reveals them one by one with a typewriter effect, followed by a fade-in book CTA.

**Tech Stack:** Astro, Tailwind CSS, vanilla JS

Replace the entire file with:

```astro
---
import BaseLayout from "../layouts/BaseLayout.astro";
---

<BaseLayout title="404 — Page Not Found" description="Page not found.">
  <div class="flex flex-col items-center justify-center py-16">
    <!-- Terminal block -->
    <div
      id="terminal"
      class="w-full max-w-xl rounded-lg border border-white/10 bg-neutral-900/50 p-6 font-mono text-sm text-green-400"
    >
      <div class="terminal-line"><span class="prompt text-neutral-500">&gt; </span><span class="typed">Searching for "<span id="path-slot">/unknown</span>"...</span></div>
      <div class="terminal-line"><span class="prompt text-neutral-500">&gt; </span><span class="typed">Scanning 42 published articles...</span></div>
      <div class="terminal-line"><span class="prompt text-neutral-500">&gt; </span><span class="typed">Checking redirects...</span></div>
      <div class="terminal-line text-red-400"><span class="prompt text-neutral-500">&gt; </span><span class="typed">Page not found.</span></div>
      <div class="terminal-line empty">&nbsp;</div>
      <div class="terminal-line"><span class="prompt text-neutral-500">&gt; </span><span class="typed">Diagnosis: This page was never built.</span></div>
      <div class="terminal-line"><span class="prompt text-neutral-500">&gt; </span><span class="typed">Probably someone vibe-coded the URL.</span></div>
      <div class="terminal-line empty">&nbsp;</div>
      <div class="terminal-line"><span class="prompt text-neutral-500">&gt; </span><span class="typed">Recommended fix:</span></div>
      <div class="terminal-line text-white font-bold"><span class="prompt text-neutral-500">&gt; </span><span class="typed">Stop guessing. Start shipping.</span></div>
      <span id="cursor" class="hidden w-2 h-4 bg-green-400 animate-pulse ml-0.5 align-middle"></span>
    </div>

    <!-- Book promo CTA (visible by default for no-JS, JS hides then fades in) -->
    <a
      id="book-cta"
      href="/prompt-contract"
      class="group mt-8 flex items-center gap-4 rounded-xl border border-white/10 bg-white/[0.02] p-4 transition-all hover:border-sky-400/40 no-underline max-w-xl w-full"
    >
      <img
        src="/prompt-contracts-cover.png"
        alt="Prompt Contracts — book cover"
        class="w-20 flex-shrink-0 rounded-md"
        loading="lazy"
      />
      <div>
        <span class="inline-block text-xs font-semibold uppercase tracking-wider text-sky-400 bg-sky-400/10 border border-sky-400/20 rounded-full px-2.5 py-0.5">
          New book
        </span>
        <h3 class="mt-1.5 text-base font-bold text-white transition-colors group-hover:text-sky-400">
          Prompt Contracts
        </h3>
        <p class="mt-0.5 text-xs text-neutral-400">
          Stop vibe coding. Ship real software with AI.
        </p>
      </div>
    </a>

    <!-- Back link (visible by default for no-JS) -->
    <a
      id="back-link"
      href="/blog"
      class="mt-6 text-sm text-neutral-500 transition-colors hover:text-neutral-300"
    >
      or go back to the blog
    </a>
  </div>

  <script>
    const lines = document.querySelectorAll<HTMLElement>('.terminal-line');
    const cursor = document.getElementById('cursor')!;
    const bookCta = document.getElementById('book-cta')!;
    const backLink = document.getElementById('back-link')!;
    const pathSlot = document.getElementById('path-slot')!;

    // Inject actual path
    pathSlot.textContent = window.location.pathname;

    const prefersReducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

    if (prefersReducedMotion) {
      // Everything already visible from HTML — just hide the cursor
      cursor.style.display = 'none';
    } else {
      // Hide lines, CTA, back link — then animate
      lines.forEach(l => l.style.display = 'none');
      bookCta.style.opacity = '0';
      backLink.style.opacity = '0';
      cursor.classList.remove('hidden');
      cursor.classList.add('inline-block');

      const CHAR_DELAY = 30;
      const LINE_PAUSE = 400;

      function typewriteLine(typedSpan: HTMLElement): Promise<void> {
        return new Promise(resolve => {
          const fullText = typedSpan.textContent || '';
          typedSpan.textContent = '';
          let i = 0;
          function tick() {
            if (i < fullText.length) {
              typedSpan.textContent = fullText.slice(0, i + 1);
              i++;
              setTimeout(tick, CHAR_DELAY);
            } else {
              resolve();
            }
          }
          tick();
        });
      }

      async function runSequence() {
        for (const line of lines) {
          line.style.display = '';

          if (line.classList.contains('empty')) {
            await new Promise(r => setTimeout(r, LINE_PAUSE));
            continue;
          }

          // Place cursor inside the line, after the typed span
          const typedSpan = line.querySelector('.typed') as HTMLElement;
          line.appendChild(cursor);

          await typewriteLine(typedSpan);
          await new Promise(r => setTimeout(r, LINE_PAUSE));
        }

        // Hide cursor, reveal CTA and back link
        cursor.style.display = 'none';
        bookCta.style.transition = 'opacity 300ms ease';
        bookCta.style.opacity = '1';
        backLink.style.transition = 'opacity 300ms ease 150ms';
        backLink.style.opacity = '1';
      }

      runSequence();
    }
  </script>
</BaseLayout>
```
